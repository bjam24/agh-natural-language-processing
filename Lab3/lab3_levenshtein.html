def distort_word(word):
    letter_idx = random.randint(0, len(word) - 1)
    random_letter = random.choice(list(string.ascii_letters))
    
    while word[letter_idx].lower() == random_letter.lower():
        random_letter = random.choice(list(string.ascii_letters))

    word = list(word)
    word[letter_idx] = random_letter
    
    return ''.join(word)

def distort_query(query):
    query_tokens = [token.text for token in nlp(query)]
    word_tokens = [token.text for token in nlp(query) if token.is_alpha]

    if not word_tokens:  
        return query, None
    
    token_idx = random.randint(0, len(query_tokens) - 1)

    while query_tokens[token_idx] not in word_tokens:
        token_idx = random.randint(0, len(query_tokens) - 1)

    distorted_word = distort_word(query_tokens[token_idx])
    query_tokens[token_idx] = distorted_word
    
    return ' '.join(query_tokens), distorted_word

def search_answers_ids(es, index_name, query, analyzer, text, k, fuzzy):
    body_query = {
        'query': {
            'match': {
                text: {
                    'query': query,
                    'analyzer': analyzer
                }
            }
        },
        '_source': ['id']
    }

    if fuzzy:
        body_query['query']['match'][text].update(fuzzy)
    
    response = es.search(index = index_name, body = body_query)['hits']['hits'][:k]
    return [hit['_source']['id'] for hit in response]

def compute_ncdgk(answers_ids, revelant_ids, k):
    dcg = sum(1 / math.log(i + 2, 2) for i in range(k) if int(answers_ids[i]) in revelant_ids)
    idcg = sum(1 / math.log(i + 2, 2) for i in range(min(len(revelant_ids), k)))
    
    return dcg / idcg
    
def compute_analyzer_mean_ncdgn(analyzer, text, analysed_col, k, fuzzy = None):
    ncdgk_scores = []
    
    for query_id in fiqa_pl_test_df['query-id'].unique():
        query = fiqa_pl_queries_df[str(query_id) == fiqa_pl_queries_df['_id']].iloc[0][analysed_col]
        answers_ids = search_answers_ids(es, 'fiqa_pl_index', query, analyzer, text, k, fuzzy)
        relevant_ids = fiqa_pl_test_df[query_id == fiqa_pl_test_df['query-id']]['corpus-id'].tolist()
        ncdgk_score = compute_ncdgk(answers_ids, relevant_ids, k = k)
        ncdgk_scores.append(ncdgk_score)

    return np.mean(ncdgk_scores)

from polyleven import levenshtein

def find_correction(invalid_word, frequency_dict):
    suggestions = sorted([(word, levenshtein(invalid_word, word)) for word in frequency_dict.keys()], key = lambda x: x[1])
    return suggestions[0][0]
